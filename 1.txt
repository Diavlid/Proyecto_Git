--crear un table space fisico 

CREATE TABLESPACE QUINTO DATAFILE 'C:\oraclexe\app\oracle\oradata\XE\QUINTO.DBF' SIZE 10M AUTOEXTEND ON NEXT 200K MAXSIZE 14000K
DEFAULT STORAGE (INITIAL 16K NEXT 16K MINEXTENTS 1 MAXEXTENTS 3);

--CREAR UN TABLE SPACE TEMPORAL
CREATE TEMPORARY TABLESPACE TEMP_QUINTO TEMPFILE 'C:\oraclexe\app\oracle\oradata\XE\TEMP_QUINTO.DBF' 
SIZE 500K AUTOEXTEND ON NEXT 50K MAXSIZE 600K
EXTENT MANAGEMENT LOCAL UNIFORM SIZE 100K;

--CREAR UN USUARIO

CREATE USER COMPU IDENTIFIED BY COMPU
DEFAULT TABLESPACE QUINTO
TEMPORARY TABLESPACE TEMP_QUINTO
QUOTA 1000K ON QUINTO;

--ASIGNACION DE PERMISOS AL USUARIO

GRANT CREATE SESSION TO COMPU;

--ASIGNAR ROL AL USUARIO

GRANT ROL_COMPU TO COMPU;

--CREAR UN ROL

CREATE ROLE ROL_COMPU; 

GRANT CREATE TABLE TO ROL_COMPU;

SELECT * FROM DBA_USERS
WHERE USERNAME ='COMPU';

SELECT * FROM DBA_USERS
WHERE ROLE ='COMPU';
